import pandas as pd

def find_s_algorithm(file_path):
    # Step 1: Load training data from CSV file
    data = pd.read_csv(file_path)

    print("Training data:")
    print(data)

    # Step 2: Separate attributes and class label
    attributes = data.columns[:-1]
    class_label = data.columns[-1]

    # Step 3: Initialize hypothesis with most specific values
    hypothesis = ['?' for _ in attributes]

    # Step 4: Apply Find-S algorithm
    for index, row in data.iterrows():
        if row[class_label] == 'Yes':
            for i, value in enumerate(row[attributes]):
                if hypothesis[i] == '?' or hypothesis[i] == value:
                    hypothesis[i] = value
                else:
                    hypothesis[i] = '?'

    return hypothesis


# Step 5: Specify the CSV file path
file_path = 'training_data.csv'

# Step 6: Run the Find-S algorithm
hypothesis = find_s_algorithm(file_path)

# Step 7: Display the final hypothesis
print("\nThe final hypothesis is:", hypothesis)
